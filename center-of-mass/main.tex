\begin{align*}
\vec{r} \left( m_1 + m_2 + \cdots + m_n \right) &= \vec{r}_1m_1 + \vec{r}_2m_2 + \cdots + \vec{r}_nm_n\\[4mm]
\vec{r}\left(\sum_{i=0}^{i=n} m_i\right)  &= \sum_{i=0}^{i=n} (\vec{r}_i m_i )\\[5mm]
\end{align*}
